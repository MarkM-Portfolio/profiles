facebook.getFriends = function (caller) {

  var api = null;

  caller.toggleLoading(true);
  
  FB_RequireFeatures(["Api"], function() {
    //FB.FBDebug.isEnabled = true;
    //FB.FBDebug.logLevel = 4;

    // Create an ApiClient object, passing app’s api key and
    // a site relative url to xd_receiver.htm
    api = new FB.ApiClient('d473d2a9a7e142d85c560be21f5c04ca', '/widgets/xd_receiver.htm', null);

    // require user to login
    api.requireLogin(function(exception) {

        console.log("Current Facebook user id is " + api.get_session().uid);
        // Get friends list
        api.friends_get(null, function(result, exception) {
          api.users_getInfo(result, "first_name, last_name, current_location", function(result) {
            caller.friends = result;
            console.log(result);
            for (i=0; i< caller.friends.length; ++i) {
              caller.friends[i].name = caller.friends[i].first_name + " " + caller.friends[i].last_name;
              caller.friends[i].uid = null;
            }
            caller.friendIter = 0;
            caller.searchProfiles(caller.friends[0], caller, 'name');
          });
        });
    });

  });

}
